# ============================================================================= # Stim Backend — Photon-Aware Noise v9 # NEW v9: # - OrnsteinUhlenbeckProcess extended for thermal photon fluctuations # - TelegraphNoiseProcess: random two-level system (TLS) flip noise # - PinkNoiseGenerator: 1/f correlated noise # - sample_syndromes: modulates readout + Z errors with photon/TLS state # - photon_estimate exposed in observation vector # - compile_detector_error_model: real DEM via Stim # - sample_dem_syndromes: DEM-aligned multi-round sampling # ============================================================================= class OrnsteinUhlenbeckProcess: """Mean-reverting OU process — used for mobile noise AND thermal photons.""" def __init__(self, mu: float = 0.0, theta: float = 0.15, sigma: float = 0.02, dt: float = 1.0): self.mu = mu; self.theta = theta; self.sigma = sigma; self.dt = dt self.state = float(mu) def step(self) -> float: dx = (self.theta * (self.mu - self.state) * self.dt + self.sigma * math.sqrt(self.dt) * random.gauss(0, 1)) self.state += dx return self.state def reset(self): self.state = float(self.mu) class TelegraphNoiseProcess: """ NEW v9: Random Telegraph Process modelling TLS (two-level system) jumps. 12. Environments — QuantumPhoneEnv, Vector, Photon-Enriched Observations

# ============================================================================= # Environments # NEW v9: # - QuantumPhoneEnv: enriched observation space # obs = [syndrome_history (S×H), leakage_map (Q), photon_estimate, # battery_voltage, temp_proxy] # - QuantumPhoneEnvVector: same enrichment for vectorised envs # ============================================================================= class QuantumPhoneEnv: """ Single-environment RL wrapper for a stabilizer code. v9: enriched observation includes syndrome history, leakage map, photon estimate, battery voltage proxy, and temperature proxy. """ def __init__(self, config: EdgeQECConfig, code: StabilizerCode): self.cfg = config; self.code = code self.noise = config.noise; self.rl = config.rl self.backend = StimBackend(code, config.noise) self.lcd = LocalClusteringDecoder(code) self.reward_norm = RewardNormalizer() # Observation dimensions self.n_s = code.num_stabilizers self.n_q = code.num_qubits self.h_rounds = self.rl.syndrome_history_rounds # History buffer self._syn_history = np.zeros((self.h_rounds, self.n_s), dtype=np.float32) # state_dim: history + leakage_map + photon_est + battery + temp self.state_dim = (self.n_s * self.h_rounds # syndrome history + self.n_q # leakage map + (3 if self.rl.expose_photon_obs else 0)) # photon_est, battery, temp_proxy self._step_count = 0 self._current_syndrome = np.zeros(self.n_s, dtype=np.float32) def _build_obs(self) -> np.ndarray: """Assemble the enriched observation vector.""" history_flat = self._syn_history.flatten() leakage_map = self.backend.leakage.leaked.astype(np.float32) if self.rl.expose_photon_obs: photon_est = np.array([self.backend.photon_estimate], dtype=np.float32) battery = np.array([np.clip(1.0 - self._step_count * 1e-4, 0, 1)], dtype=np.float32) temp_proxy = np.array([0.02 * self.backend.photon_estimate], dtype=np.float32) return np.concatenate([history_flat, leakage_map, photon_est, battery, temp_proxy]) return np.concatenate([history_flat, leakage_map]) def reset(self) -> np.ndarray: self._syn_history[:] = 0 self.backend.leakage.leaked[:] = False self.backend.photon_ou.reset() self.backend.ou_mobile.reset() self._step_count = 0 syndromes = self.backend.sample_syndromes(rounds=1, num_shots=1) self._current_syndrome = syndromes[0] # Shift history self._syn_history = np.roll(self._syn_history, 1, axis=0) self._syn_history[0] = self._current_syndrome return self._build_obs() def step(self, action: int) -> Tuple[np.ndarray, float, bool, Dict]: self._step_count += 1 corr = np.zeros(self.n_q, dtype=int) if action > 0 and action - 1 < self.n_q: corr[action - 1] = 1 ex, ez = self.code.random_error(self.noise.depolarizing_error_rate) ex = (ex + corr) % 2 logical_err = int(np.dot(self.code.logical_z(), ex) % 2) syndrome_weight = float(self._current_syndrome.sum()) reward = -1.0 if logical_err else +1.0 reward -= self.rl.latency_penalty_coef * syndrome_weight reward = self.reward_norm(reward, logical_err == 1) syndromes = self.backend.sample_syndromes(rounds=1, num_shots=1) self._current_syndrome = syndromes[0] self._syn_history = np.roll(self._syn_history, 1, axis=0) self._syn_history[0] = self._current_syndrome done = bool(logical_err) obs = self._build_obs() info = { "logical_error": logical_err, "syndrome_weight": syndrome_weight, "photon_estimate": self.backend.photon_estimate, "leakage_count": int(self.backend.leakage.leaked.sum()), } return obs, reward, done, info def get_state_dim(self) -> int: return self.state_dim # For compatibility with code.get_state_dim() @property def code(self): return self._code @code.setter def code(self, c): self._code = c class QuantumPhoneEnvVector: """ Vectorised environment wrapper (32–128 parallel envs). v9: uses enriched QuantumPhoneEnv observation space. 13. Training Pipeline

# ============================================================================= # Training Pipeline # NEW v9: # - Passes photon_estimate + leakage_count to DreamerV3Agent.update() # - World model warm-start before RL training # - Passes real_device_buf to DreamerV3Agent if trace_path provided # ============================================================================= def _make_agent(config: EdgeQECConfig, obs_dim: int, action_dim: int, algorithm: str, code: StabilizerCode): rl = config.rl; hw = config.hardware if algorithm == "dqn": return DQNAgent(obs_dim, action_dim, rl, hw) elif algorithm in ("ppo", "a2c"): return PPOAgent(obs_dim, action_dim, rl) elif algorithm == "sac": return SACAgent(obs_dim, action_dim, rl) elif algorithm == "td3": return TD3Agent(obs_dim, action_dim, rl) elif algorithm == "hierarchical": return HierarchicalDQNAgent(obs_dim, action_dim, rl, hw) elif algorithm == "dreamer": lcd = LocalClusteringDecoder(code) return DreamerV3Agent(obs_dim, action_dim, rl, lcd_decoder=lcd) elif algorithm == "hierarchical_dreamer": return HierarchicalDreamerAgent(obs_dim, action_dim, num_patches=4, rl_cfg=rl) elif algorithm == "spa_marl": return SPAMARLAgent(obs_dim, action_dim, rl) elif algorithm == "hybrid": base = DQNAgent(obs_dim, action_dim, rl, hw) return HybridRLDecoder(code, base) else: return DQNAgent(obs_dim, action_dim, rl, hw) def train_agent(config: EdgeQECConfig, episodes: int = 1000, quantum_code: Optional[StabilizerCode] = None, algorithm: str = "dqn", n_envs: int = 1, reward_mode: str = "logical", verbose: bool = True): set_seed(config.seed) code = quantum_code or SurfaceCode(3) env = QuantumPhoneEnv(config, code) obs_dim = env.state_dim action_dim = code.get_action_dim() agent = _make_agent(config, obs_dim, action_dim, algorithm, code) #
