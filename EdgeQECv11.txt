""" return compute_logical_z(self.check_matrix_z(), self.num_qubits) def get_state_dim(self, rounds: int = 1) -> int: return self.num_stabilizers * rounds + self.num_qubits def get_action_dim(self) -> int: return self.num_qubits * 3 + 1 def compute_logical_z(Hz: np.ndarray, n_qubits: int) -> np.ndarray: """ NEW v11: Compute a Z-logical operator as a representative of the kernel of Hz over GF(2) that is not in the row space of Hz. Uses Gaussian elimination (GF(2)) to find a low-weight coset representative. If no non-trivial kernel element is found (code has no logical qubits), returns Hz[0] as a fallback. """ m, n = Hz.shape # Augment Hz with identity to track row operations A = np.hstack([Hz.copy().astype(np.uint8), np.eye(m, dtype=np.uint8)]) pivot_cols = [] row = 0 for col in range(n): # Find pivot candidates = np.where(A[row:, col] == 1)[0] if len(candidates) == 0: continue swap = candidates[0] + row A[[row, swap]] = A[[swap, row]] for r2 in range(m): if r2 != row and A[r2, col] == 1: A[r2] = (A[r2] + A[row]) % 2 pivot_cols.append(col) row += 1 if row == m: break # Free (non-pivot) columns correspond to logical operators all_cols = set(range(n)) free_cols = sorted(all_cols - set(pivot_cols)) if not free_cols: logger.warning("[compute_logical_z] No free columns; returning Hz[0]") return Hz[0].astype(int) # Build a logical operator from the first free column logical = np.zeros(n, dtype=int) logical[free_cols[0]] = 1 for i, pc in enumerate(pivot_cols): if i < A.shape[0] and A[i, free_cols[0]] == 1: logical[pc] = 1 return logical # ── Fixed Tanner-graph face construction (NEW v11) ─────────────────────────── def _surface_tanner_faces(d: int) -> Tuple[List[List[int]], List[List[int]]]: """ Build X-stabiliser and Z-stabiliser face lists for the d×d rotated surface code using a SINGLE unified checkerboard sweep. Layout: Qubits are indexed row-major: qubit(r, c) = r*d + c, 0 ≤ r,c < d. Face centres: A face centred at (r + 0.5, c + 0.5) is an X-face when (r+c) is even, and a Z-face when (r+c) is odd. The face touches the four corner qubits: (r,c), (r,c+1), (r+1,c), (r+1,c+1). Boundary qubits (out of range) are silently dropped, giving weight-2 faces. """ def qubit(r: int, c: int) -> Optional[int]: if 0 <= r < d and 0 <= c < d: return r * d + c return None x_faces: List[List[int]] = [] z_faces: List[List[int]] = [] # ── Bulk faces ─────────────────────────────────────────────────────────── for r in range(d - 1): for c in range(d - 1): corners = [qubit(r, c), qubit(r, c + 1), qubit(r + 1, c), qubit(r + 1, c + 1)] face = [q for q in corners if q is not None] if (r + c) % 2 == 0: x_faces.append(face) else: z_faces.append(face) # ── Boundary X faces (top and bottom, weight-2) ────────────────────────── # Top boundary: centres at (-0.5, c+0.5) for c in range(d-1), (r+c) parity for c in range(d - 1): if (0 + c) % 2 == 0: # top row r=0, face at r=-0.5 face = [q for q in [qubit(0, c), qubit(0, c + 1)] if q is not None] if len(face) == 2: x_faces.append(face) # Bottom boundary: centres at (d-0.5, c+0.5) for c in range(d - 1): if ((d - 1) + c) % 2 == 0: face = [q for q in [qubit(d - 1, c), qubit(d - 1, c + 1)] if q is not None] if len(face) == 2: x_faces.append(face) # ── Boundary Z faces (left and right, weight-2) ────────────────────────── # Left boundary: centres at (r+0.5, -0.5) for r in range(d-1), odd parity for r in range(d - 1): if (r + 0) % 2 == 1: # left col c=0, face at c=-0.5 face = [q for q in [qubit(r, 0), qubit(r + 1, 0)] if q is not None] if len(face) == 2: z_faces.append(face) # Right boundary: centres at (r+0.5, d-0.5) for r in range(d - 1): if (r + (d - 1)) % 2 == 1: face = [q for q in [qubit(r, d - 1), qubit(r + 1, d - 1)] if q is not None] if len(face) == 2: z_faces.append(face) return x_faces, z_faces class SurfaceCode(StabilizerCode): """ Rotated surface code with FIXED Tanner-graph Hx/Hz (NEW v11).
